#include <stdio.h>
#include <string.h>
    
    int l;
    
    void a(char *s){
        int b = 5;
        int length = strlen(s);
        
        printf("%*s%*c",((b - length) >> 1) + length //string length + padding spaces
            , s
            , (( b - length) >> 1) + (( b - length) & 1)
            , ' '
        );
    }
    
    void space(int spa){
        for(l = 1; l <=spa; l++){
          a(" ");
        }
    }
    
    void valor(int val){
        for(l = 1; l <= val; l++){
           a(" * ");
        }
        printf("\n");
    }
